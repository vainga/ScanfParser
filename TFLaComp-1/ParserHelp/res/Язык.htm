Минимальные варианты:

scanf("");

С пустой строкой формата и аргументами:

scanf("", &x);
scanf("", &x, &y);
scanf("", &variable123);

Со спецификаторами формата:

scanf("%d");
scanf("%s", &name);
scanf("%d", &x);
scanf("%d %f", &x, &y);

С обычными символами в формате:

scanf("Hello");
scanf("Введите число:", &x);

Сложные примеры:

scanf("%d %s %f", &number, &text, &decimal);
scanf("Имя: %s Возраст: %d", &name, &age);
scanf("", &var1, &var2, &var3);
